% Slides with Only an Image


% 1. Title of page
% 2. Size of image (width)
% 3. image path
% 4. description of image
\newcommand{\pageWithImageWidth}[4]{
    \begin{frame}
      \frametitle{#1}
      \includegraphics[width=#2]{#3}
     {\footnotesize #4 }
     \end{frame}
   }

% Slides with Image and Text to the side
% 1. Frametitle
% 2. height
% 3. width
% 4. image path
% 5. text afterwards
% 6. image description
   
\newcommand{\pageWithImageAndTextRight}[6]{
\begin{frame}
  \frametitle{#1}

  \begin{tabular}{cl}  
    \begin{tabular}{c}
      \includegraphics[height=#2, width=#3]{#4}\\
      {\footnotesize #6}
    \end{tabular}
    & \begin{tabular}{l}
      \parbox{0.5\linewidth}{%  change the parbox width as appropiate
      #5
    }

    \end{tabular}  \\
    \end{tabular}
    

\end{frame}
}
